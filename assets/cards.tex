% ---------- CARD MACROS ----------

\newcommand{\cardfront}[3]{%
\thispagestyle{empty}%

\begin{tikzpicture}[remember picture,overlay]

  % ---- FORCE PAGE BOUNDING BOX ----
  \path[use as bounding box]
    (current page.south west)
    rectangle
    (current page.north east);   

  % work in mm from now on
  \begin{scope}[x=1mm,y=1mm]

    % ---- constants ----
    \def\bleed{3.175}    % 1/8"
    \def\cardW{63}
    \def\cardH{88}
    \def\safe{4}

    \def\nameH{14}
    \def\typeH{10}
    \def\descH{52}

    \pgfmathsetmacro{\innerW}{\cardW - 2*\safe}

    % page origin
    \coordinate (O) at (current page.south west);

    % ---- trim outline ----
    \draw[line width=0.4mm]
      ($(O)+(\bleed,\bleed)$)
      rectangle ++(\cardW,\cardH);

    % ---- NAME ----
    \draw
      ($(O)+(\bleed+\safe,\bleed+\cardH-\safe-\nameH)$)
      rectangle ++(\innerW,\nameH);

    \node[align=center, text width=\innerW mm]
      at ($(O)+(\bleed+0.5*\cardW,\bleed+\cardH-\safe-0.5*\nameH)$)
      { \centering
        \sloppy
        \exhyphenpenalty=10000
        \hyphenpenalty=10000
        \titlefont\fontsize{16}{20}\selectfont #1};    

    % ---- TYPE ----
    \draw
      ($(O)+(\bleed+\safe,\bleed+\cardH-\safe-\nameH-\typeH)$)
      rectangle ++(\innerW,\typeH);

    \node[align=center]
      at ($(O)+(\bleed+0.5*\cardW,\bleed+\cardH-\safe-\nameH-0.5*\typeH)$)
      {\itshape #2};

    % ---- DESCRIPTION ----
    \draw
      ($(O)+(\bleed+\safe,\bleed+\safe)$)
      rectangle ++(\innerW,\descH);

    \node[
      align=center,
      text width=\innerW mm
    ]
      at ($(O)+(\bleed+0.5*\cardW,\bleed+\safe+0.5*\descH)$)
      {\raggedright\hyphenpenalty=20000\fontsize{10}{13}\selectfont #3};

  \end{scope}
\end{tikzpicture}

\newpage
}
