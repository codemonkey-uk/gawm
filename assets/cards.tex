% ---------- CARD MACROS ----------

\newcommand{\safeText}[1]{%
  \if\relax\detokenize{#1}\relax
    ~%
  \else
    #1%
  \fi
}

\newcommand{\boxedtext}[5]{%
  \draw[line width=0.3mm] (#1,#2) rectangle ++(#3,#4);
  \node[
    align=center,
    text width=#3 mm,
    inner sep=2mm
  ] at (#1 + #3/2, #2 + #4/2) {#5};
}

\newcommand{\cardfront}[3]{%
  \clearpage
  \thispagestyle{empty}%
  \shipout\hbox to \paperwidth{%
    \vbox to \paperheight{%
      \vfil\hfil
      \begin{tikzpicture}[x=1mm,y=1mm]

        % --- Dimensions ---
        \pgfmathsetmacro{\W}{63}
        \pgfmathsetmacro{\H}{88}
        \pgfmathsetmacro{\safe}{4}
        \pgfmathsetmacro{\nameH}{12}
        \pgfmathsetmacro{\subH}{10}
        \pgfmathsetmacro{\innerW}{\W - 2*\safe}
        \pgfmathsetmacro{\descH}{\H - 2*\safe - \nameH - \subH}

        % --- Card outline ---
        \draw[line width=0.3mm] (0,0) rectangle (\W,\H);

        % --- Name ---
        \boxedtext
          {\safe}
          {\H - \safe - \nameH}
          {\innerW}
          {\nameH}
          {{\Large #1}}

        % --- Subtype ---
        \boxedtext
          {\safe}
          {\H - \safe - \nameH - \subH}
          {\innerW}
          {\subH}
          {{\itshape #2}}

        % --- Description ---
        \boxedtext
          {\safe}
          {\safe}
          {\innerW}
          {\descH}
          {{\small #3}}

      \end{tikzpicture}
      \hfil\vfil
    }%
  }%
}
