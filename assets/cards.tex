% --- Art Deco geometry constants ---
\newcommand{\decoInset}{2pt}
\newcommand{\decoStep}{3pt}
\newcommand{\decoDotRadius}{1pt}

% #1 = horizontal run length
% #2 = vertical run length
% #3 = draw horizontal end dot? (0/1)
% #4 = draw vertical end dot?   (0/1)
\newcommand{\artDecoCorner}[4]{%

  % Outer frame lines (relative length)
  \draw (0,#2) -- (0,0) -- (#1,0);

  % Stepped inset (absolute depth)
  \draw
    (\decoInset,#2-\decoInset) --
    (\decoInset,\decoInset) --
    (#1-\decoInset,\decoInset);

  % Circular accents (corner)
  \fill
    (\decoInset+\decoStep,\decoInset+\decoStep)
    circle (\decoDotRadius);

  % Circular accent (line end v)
  \ifnum#4=1
    \fill[yshift=\decoStep]
        (0,#2)
        circle (\decoDotRadius);
  \fi

  % Circular accent (line end h)
  \ifnum#3=1
    \fill[xshift=\decoStep]
        (#1,0)
        circle (\decoDotRadius);
  \fi
}


\newcommand{\ornateframe}[4][]{%
  % #1 = optional TikZ style
  % #2 = origin
  % #3 = width
  % #4 = height

  \pgfmathsetmacro{\cornerFrac}{0.18}
  \pgfmathsetmacro{\cornerX}{max(#3 * \cornerFrac, 2*\decoInset + \decoStep + 4pt)}
  \pgfmathsetmacro{\cornerY}{max(#4 * \cornerFrac, 2*\decoInset + \decoStep + 4pt)}

  \pgfmathsetmacro{\dotH}{ifthenelse(\cornerX > 0.45*#3, 0, 1)}
  \pgfmathsetmacro{\dotV}{ifthenelse(\cornerY > 0.45*#4, 0, 1)}

  \begin{scope}[shift={#2}, #1]

    % bottom-left
    \artDecoCorner{\cornerX}{\cornerY}{\dotH}{\dotV}

    % bottom-right
    \begin{scope}[shift={(#3,0)}, xscale=-1]
      \artDecoCorner{\cornerX}{\cornerY}{\dotH}{\dotV}
    \end{scope}

    % top-left
    \begin{scope}[shift={(0,#4)}, yscale=-1]
      \artDecoCorner{\cornerX}{\cornerY}{\dotH}{\dotV}
    \end{scope}

    % top-right
    \begin{scope}[shift={(#3,#4)}, xscale=-1, yscale=-1]
      \artDecoCorner{\cornerX}{\cornerY}{\dotH}{\dotV}
    \end{scope}

  \end{scope}
}

\newcommand{\tokenbox}[3]{%
  % #1 = x
  % #2 = y
  % #3 = content

  \ifx#3G
  {
    % guilt logic
    \fill[RichRed]
      ($(O)+(#1,#2)$)
      rectangle
      ($(O)+(#1+\tokenSize,#2+\tokenSize)$);
    % centered content
    \node at ($(O)+(#1+0.5*\tokenSize,#2+0.5*\tokenSize)$)
    {
      \begin{minipage}[c][\tokenSize mm][c]{\tokenSize mm}
        \centering
        \includegraphics[
          width=0.65\linewidth,
          height=0.65\linewidth,
          keepaspectratio
        ]{guilt.png}

      \end{minipage}
    };
  }      
  \else\ifx#3I
  {
    % innocence logic
    \fill[RichBlue]
      ($(O)+(#1,#2)$)
      rectangle
      ($(O)+(#1+\tokenSize,#2+\tokenSize)$);
    % centered content
    \node at ($(O)+(#1+0.5*\tokenSize,#2+0.5*\tokenSize)$)
    {
      \begin{minipage}[c][\tokenSize mm][c]{\tokenSize mm}
        \centering
        \includegraphics[
          width=0.65\linewidth,
          height=0.65\linewidth,
          keepaspectratio
        ]{innocence.png}

      \end{minipage}
    };      
  }
  \else
    % treat as text
    \node[align=center]
      at ($(O)+(#1+0.5*\tokenSize,#2+0.5*\tokenSize)$)
      {\parbox{\tokenSize mm}{\centering \tokenfont #3}};
  \fi\fi

  \ornateframe[thick]
    {($(O)+(#1,#2)$)}
    {\tokenSize}
    {\tokenSize}
}

\infolist
{Game Overview}
{
\item \textbf{Act I} - One scene per player.
\item \textbf{Extra Scene} - Decide the victim \& Introduce a new character.
\item \textbf{Murder Scene} - Body is found.
\item \textbf{Act II} - One scene per player.
\item \textbf{The Twist} - Discard \& draw.
\item \textbf{Act III} - Two scenes per player.
\item \textbf{The Accusation} - The Most Innocent player accuses.
\item \textbf{The Guilty} - Guilt is determined.
\item \textbf{Epilogue} - One per player.
}

\infolist
{Game Setup}
{
\item Shuffle all decks.
\item For each player, add a \textbf{guilt token} labelled \textbf{0}, \textbf{1}, \textbf{2} and \textbf{3} to a central pile and shuffle with numbers \textbf{face-down}.
\item Repeat for \textbf{innocence tokens}.
\item Each player draws \textbf{2 alias cards} and chooses one to play.
\item Each player draws \textbf{3 wildcards}, \textbf{3 relationships}, \textbf{3 motives} and \textbf{3 objects}. Don't reveal your cards.
}
