% ---------- CARD MACROS ----------

\newcommand{\cardfront}[4]{%
\thispagestyle{empty}%

\begin{tikzpicture}[remember picture,overlay]

  % ---- force page bounding box ----
  \path[use as bounding box]
    (current page.south west)
    rectangle
    (current page.north east);

  \begin{scope}[x=1mm,y=1mm]

    % ---- constants ----
    \def\bleed{3.175}
    \def\cardW{63}
    \def\cardH{88}
    \def\safe{4}

    % box heights (tune later)
    \def\catH{8}
    \def\typeH{10}
    \def\titleH{14}
    \def\descH{46}

    \pgfmathsetmacro{\innerW}{\cardW - 2*\safe}

    \coordinate (O) at (current page.south west);

    % ---- trim outline ----
    \draw[line width=0.4mm]
      ($(O)+(\bleed,\bleed)$)
      rectangle ++(\cardW,\cardH);

    % ---- CATEGORY ----
    \draw
      ($(O)+(\bleed+\safe,\bleed+\cardH-\safe-\catH)$)
      rectangle ++(\innerW,\catH);

    \node[align=center]
      at ($(O)+(\bleed+0.5*\cardW,\bleed+\cardH-\safe-0.5*\catH)$)
      {\fontsize{9}{11}\selectfont #1};

    % ---- TITLE ----
    \draw
      ($(O)+(\bleed+\safe,\bleed+\cardH-\safe-\catH-\typeH)$)
      rectangle ++(\innerW,\typeH);

    \node[
      align=center,
      text width=\innerW mm
    ]
      at ($(O)+(\bleed+0.5*\cardW,\bleed+\cardH-\safe-\catH-0.5*\typeH)$)
      {\itshape\fontsize{12}{14}\selectfont #3};

    % ---- SUBTYPE ----
    \draw
      ($(O)+(\bleed+\safe,\bleed+\cardH-\safe-\catH-\typeH-\titleH)$)
      rectangle ++(\innerW,\titleH);

    \node[
      align=center,
      text width=\innerW mm
    ]
      at ($(O)+(\bleed+0.5*\cardW,\bleed+\cardH-\safe-\catH-\typeH-0.5*\titleH)$)
      {\titlefont\fontsize{18}{20}\selectfont #2};

    % ---- DESCRIPTION ----
    \draw
      ($(O)+(\bleed+\safe,\bleed+\safe)$)
      rectangle ++(\innerW,\descH);

    \node[
      align=center,
      text width=\innerW mm
    ]
      at ($(O)+(\bleed+0.5*\cardW,\bleed+\safe+0.5*\descH)$)
      {\fontsize{10}{13}\selectfont #4};

  \end{scope}
\end{tikzpicture}

\newpage
}
