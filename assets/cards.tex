% ---------- CARD MACRO ----------

\newcommand{\drawcard}[3]{%
  % Card outline (trim size)
  \draw (0,0) rectangle (6.3,8.8);

  % Title
  \node[anchor=north west,font=\titlefont\Large] at (0.3,8.5) {#1};

  % Subtype
  \node[anchor=north west,font=\itshape] at (0.3,8.0) {#2};

  % Description
  \node[anchor=north west,text width=5.7cm] at (0.3,7.5) {%
    \adjustbox{max height=4.5cm, keepaspectratio}{%
      \parbox{5.7cm}{\small #3}%
    }%
  };
}

\newcommand{\card}[3]{%
  \begin{tikzpicture}[x=1cm,y=1cm]
    \drawcard{#1}{#2}{#3}
  \end{tikzpicture}%
}

\newcommand{\cardfront}[3]{%
  \noindent
  \begin{tikzpicture}[x=1cm,y=1cm]
    % shift origin by bleed (3.175mm = 0.3175cm)
    \begin{scope}[shift={(0.3175,0.3175)}]
      \drawcard{#1}{#2}{#3}
    \end{scope}
  \end{tikzpicture}%
}
