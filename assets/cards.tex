% GAWM Print Edition Card Template, to be compiled with XeLaTeX
% following drivethrupartners Specifications for Print Cards:
%
% Create a single PDF for each card deck.
% First page is the card back (of card 1), second page is that card’s face, 
%   third page is the card back (of card 2), fourth page is that card’s face, and so on throughout the deck. 
%   The order of the pages in the PDF will be the order of the deck’s printed cards.
% No common border color between backs/faces required.
% Images contained in your PDF must be 300 dpi.
% The physical size of the image(s) should at least be the size of the card. 
%   If your image is smaller than that and you upsize the image, your card images may become pixelated.
% Files must be CMYK with your ink coverage not exceeding 240%. 
%   (A CMYK mix for a vibrant and rich black is C60%, M40%, Y40%, K100%.)
% Images used in layout should not be png, png files can not be CMYK.
%   Please use tif, psd, or if you need to jpg.
% A bleed of 1/8” (.125”) on all four sides is required. 
%   If your art does not bleed, bleed a white or black border from the safety to the bleed edge.
% Do not place any important art or text closer than 1/8” (.125”) from any finished edge.
% Do not set a die-cut for the round corners into the layout.
% Export as PDF/X-1a:2003 with all fonts embedded. (Do not include printer/crop marks.)
% The finished size is 2.5” x 3.5” for U.S. 
% Poker-sized cards or 63mm x 88mm for Euro Poker-sized cards. 
% But you must add a bleed of 1/8” on all four sides. 
% Final layout size: 2.75” x 3.75” (U.S. Poker) or 2.73” x 3.71” (Euro Poker)

% --- Art Deco geometry constants ---
\newcommand{\decoInset}{2pt}
\newcommand{\decoStep}{3pt}
\newcommand{\decoDotRadius}{1pt}

% #1 = horizontal run length
% #2 = vertical run length
% #3 = draw horizontal end dot? (0/1)
% #4 = draw vertical end dot?   (0/1)
\newcommand{\artDecoCorner}[4]{%

  % Outer frame lines (relative length)
  \draw (0,#2) -- (0,0) -- (#1,0);

  % Stepped inset (absolute depth)
  \draw
    (\decoInset,#2-\decoInset) --
    (\decoInset,\decoInset) --
    (#1-\decoInset,\decoInset);

  % Circular accents (corner)
  \fill
    (\decoInset+\decoStep,\decoInset+\decoStep)
    circle (\decoDotRadius);

  % Circular accent (line end v)
  \ifnum#4=1
    \fill[yshift=\decoStep]
        (0,#2)
        circle (\decoDotRadius);
  \fi

  % Circular accent (line end h)
  \ifnum#3=1
    \fill[xshift=\decoStep]
        (#1,0)
        circle (\decoDotRadius);
  \fi
}


\newcommand{\ornateframe}[4][]{%
  % #1 = optional TikZ style
  % #2 = origin
  % #3 = width
  % #4 = height

  \pgfmathsetmacro{\cornerFrac}{0.18}
  \pgfmathsetmacro{\cornerX}{max(#3 * \cornerFrac, 2*\decoInset + \decoStep + 4pt)}
  \pgfmathsetmacro{\cornerY}{max(#4 * \cornerFrac, 2*\decoInset + \decoStep + 4pt)}

  \pgfmathsetmacro{\dotH}{ifthenelse(\cornerX > 0.45*#3, 0, 1)}
  \pgfmathsetmacro{\dotV}{ifthenelse(\cornerY > 0.45*#4, 0, 1)}

  \begin{scope}[shift={#2}, #1]

    % bottom-left
    \artDecoCorner{\cornerX}{\cornerY}{\dotH}{\dotV}

    % bottom-right
    \begin{scope}[shift={(#3,0)}, xscale=-1]
      \artDecoCorner{\cornerX}{\cornerY}{\dotH}{\dotV}
    \end{scope}

    % top-left
    \begin{scope}[shift={(0,#4)}, yscale=-1]
      \artDecoCorner{\cornerX}{\cornerY}{\dotH}{\dotV}
    \end{scope}

    % top-right
    \begin{scope}[shift={(#3,#4)}, xscale=-1, yscale=-1]
      \artDecoCorner{\cornerX}{\cornerY}{\dotH}{\dotV}
    \end{scope}

  \end{scope}
}

\newcommand{\tokenbox}[4]{%
  % #1 = x
  % #2 = y
  % #3 = content type
  % #4 = content

  % test if #4 is empty by detokenizing
  \if\relax\detokenize{#4}\relax
    \def\tokenisempty{1}
  \else
    \def\tokenisempty{0}
  \fi

  \def\cPad{2}

  \ifx#3G
  {
    % guilt token back content
    \fill[RichRed]
      ($(O)+(#1-\cPad,#2-\cPad)$)
      rectangle
      ($(O)+(#1+\tokenSize+\cPad,#2+\tokenSize+\cPad)$);

    % show either icon+label or custom text
    \ifnum\tokenisempty=1
      % empty: icon+label
      % icon
      \node at ($(O)+(#1+0.5*\tokenSize,#2+0.6*\tokenSize)$)
      {
        \begin{minipage}[c][\tokenSize mm][c]{\tokenSize mm}
          \centering
          \includegraphics[
            width=0.5\linewidth,
            height=0.5\linewidth,
            keepaspectratio
          ]{guilt.pdf}
        \end{minipage}
      };

      % label
      \node[align=center]
        at ($(O)+(#1+0.5*\tokenSize,#2+0.175*\tokenSize)$)
        {\parbox{\tokenSize mm}{\centering \tokenSmallFont GUILT}};
    \else
      % nonempty: custom text
      \node[align=center]
        at ($(O)+(#1+0.5*\tokenSize,#2+0.5*\tokenSize)$)
        {\parbox{\tokenSize mm}{\centering \tokenfont #4}};
    \fi

    % border
    \ornateframe[thick]
      {($(O)+(#1,#2)$)}
      {\tokenSize}
      {\tokenSize}
  }
  \else\ifx#3I
  {
    % innocence token back content
    \fill[RichBlue]
      ($(O)+(#1-\cPad,#2-\cPad)$)
      rectangle
      ($(O)+(#1+\tokenSize+\cPad,#2+\tokenSize+\cPad)$);

    % show either icon+label or custom text
    \ifnum\tokenisempty=1
      % empty: icon+label
      % icon
      \node at ($(O)+(#1+0.5*\tokenSize,#2+0.6*\tokenSize)$)
      {
        \begin{minipage}[c][\tokenSize mm][c]{\tokenSize mm}
          \centering
          \includegraphics[
            width=0.5\linewidth,
            height=0.5\linewidth,
            keepaspectratio
          ]{innocence_white.pdf}
        \end{minipage}
      };

      % label
      \node[align=center, text=white]
        at ($(O)+(#1+0.5*\tokenSize,#2+0.175*\tokenSize)$)
        {\parbox{\tokenSize mm}{\centering \tokenSmallFont INNOCENCE}};
    \else
      % nonempty: custom text
      \node[align=center, text=white]
        at ($(O)+(#1+0.5*\tokenSize,#2+0.5*\tokenSize)$)
        {\parbox{\tokenSize mm}{\centering \tokenfont #4}};
    \fi

    % border
    \ornateframe[thick, white]
      {($(O)+(#1,#2)$)}
      {\tokenSize}
      {\tokenSize}
  }
  \fi\fi
}

\newcommand{\tokencardgridFour}[8]{%
\begin{tikzpicture}[remember picture,overlay]

  % ---- force page bounding box ----
  \path[use as bounding box]
    (current page.south west)
    rectangle
    (current page.north east);

  \begin{scope}[x=1mm,y=1mm]

    % layout
    \def\headH{16}
    \def\bodyH{56}

    \pgfmathsetmacro{\innerW}{\cardW - 2*(\bleed+\safe)}
    \pgfmathsetmacro{\innerH}{\cardH - 2*(\bleed+\safe)}

    %  we need to fit 2 columns (1 safe space between)
    \pgfmathsetmacro{\tokenMaxW}{0.5 * (\innerW - 1*\safe)}
    %  we need to fit 2 rows (1 safe space between)
    \pgfmathsetmacro{\tokenMaxH}{0.5 * (\innerH - 1*\safe)}

    %  tokens should be square
    \pgfmathsetmacro{\tokenSize}{min(\tokenMaxW, \tokenMaxH)}

    %  center the two tokens in two columns 
    \pgfmathsetmacro{\halfW}{0.5*\cardW}
    \pgfmathsetmacro{\leftCentre}{\bleed + 0.5*\halfW}
    \pgfmathsetmacro{\rightCentre}{\bleed + \halfW + 0.5*\halfW}
    \pgfmathsetmacro{\leftO}{\leftCentre - 0.5*\tokenSize}
    \pgfmathsetmacro{\rightO}{\rightCentre - 0.5*\tokenSize}    


    \pgfmathsetmacro{\vGap}{(\cardH - 2*\tokenSize)/2}
    \pgfmathsetmacro{\bottomO}{\bleed + 0.5*\vGap}
    \pgfmathsetmacro{\vSpace}{\tokenSize + \vGap}

    \coordinate (O) at (current page.south west);
    
    % ---- bleed outline (debugging) ----
    % \draw[line width=0.4mm]
    %   ($(O)+(\bleed,\bleed)$)
    %   rectangle ++(\cardW,\cardH);

    % ---- CUT line (vertical) ----
    \draw[line width=0.4mm, dashed]
      ($(O)+(\bleed + \cardW*0.5,\bleed)$) --
      ($(O)+(\bleed + \cardW*0.5,\bleed + \cardH)$);

    % ---- CUT line (horizontal/center) ----
    \draw[line width=0.4mm, dashed]
      ($(O)+(\bleed,         \bleed + \cardH*0.5)$) --
      ($(O)+(\bleed + \cardW,\bleed + \cardH*0.5)$);

    % ---- Token BOXES ----

    % bottom left
    \tokenbox{\leftO}{\bottomO}{#1}{#2}

    % bottom right
    \tokenbox{\rightO}{\bottomO}{#3}{#4} 

    % 2nd bottom left
    \tokenbox{\leftO}{\bottomO + 1*\vSpace}{#5}{#6}

    % 2nd bottom right
    \tokenbox{\rightO}{\bottomO + 1*\vSpace}{#7}{#8}

  \end{scope}
\end{tikzpicture}

\newpage
}

  Use a \textbf{detail card} in your hand to come up with an idea for a scene.

  \textbf{Where} is it set?

  \textbf{Who} else is there?

  What is your character’s \textbf{motivation}?

  Then explain your setup to the group in 3 sentences or less.
  }

  \infocard
  {Playing A Scene}
  {%
  Set the scene.

  Role-Play the scene until the premise is fully explored.

  \textbf{No cliffhangers!}

  The group gives you a guilt or innocence token.

  Optionally, give a token of the opposite type to someone who shared the scene with you.
  }

  \stepcounter{perplayercards}
\ifnum\value{perplayercards}<6
\repeat

\infolist
{Game Overview}
{
\item \textbf{Act I} - One scene per player.
\item \textbf{Extra Scene} - Decide the victim \& Introduce a new character.
\item \textbf{Murder Scene} - Body is found.
\item \textbf{Act II} - One scene per player.
\item \textbf{The Twist} - Discard \& draw.
\item \textbf{Act III} - Two scenes per player.
\item \textbf{The Accusation} - The Most Innocent player accuses.
\item \textbf{The Guilty} - Guilt is determined.
\item \textbf{Epilogue} - One per player.
}

\infolist
{Game Setup}
{
\item Shuffle all decks.
\item For each player, add a \textbf{guilt token} labelled \textbf{0}, \textbf{1}, \textbf{2} and \textbf{3} to a central pile and shuffle with numbers \textbf{face-down}.
\item Repeat for \textbf{innocence tokens}.
\item Each player draws \textbf{2 alias cards} and chooses one to play.
\item Each player draws \textbf{3 wildcards}, \textbf{3 relationships}, \textbf{3 motives} and \textbf{3 objects}. Don't reveal your cards.
}
